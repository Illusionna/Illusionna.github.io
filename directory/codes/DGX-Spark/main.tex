\section{参数解释}

\begin{itemize}
    \item  \textbf{测试服务器}：ssh ai@192.168.120.32
    \item \textbf{pp}：填充 token 数量
    \item \textbf{tg}：生成 token 数量
    \item \textbf{c}：并发请求数
    \item \textbf{d}：已缓存历史上文 token 数量
    \item \textbf{pp t/s}：提示词（预填充）阶段，每秒处理多少 token
    \item \textbf{tg t/s}：解码生成阶段，每秒生成多少 token
    \item \textbf{TTFT (ms)}：从发出请求到收到第一个 token 的延迟
    \item \textbf{Total (ms)}：从请求发出，到完整响应全部生成完毕的总时间
    \item \textbf{Warm-up (ms)}：预热所需时间
    \item \textbf{Median-turn (s)}：所有对话轮次耗时的中位数
    \item \textbf{Quality}：生成内容质量
    \item \textbf{Responsiveness}：响应质量性能
\end{itemize}

\subsection{对比总结}
\begin{itemize}
    \item \textbf{综合最优}：$\bigstar$ vLLM.dflash + Q3.5-122B-A10B-INT4 + DFlash-8，质量满分 100，单流 TTFT 全场最低 1.13s（d0/c1），解码 tg 全场最高 91.7 t/s（d0/c4），预填充 pp 亦达 2279 t/s，中位轮次仅 3.3s，在 TTFT、TPS、Quality 中是延迟、吞吐场景的首选。
    \item \textbf{响应最佳}：vLLM.dev + Q3.5-122B-A10B Hybrid-INT4FP8 + MTP-2，质量 100、响应 48，TTFT 1.25s、pp 峰值 2421 t/s、tg 80.5 t/s、中位轮次 3.1s，各项指标几乎贴平 vLLM.dflash + Q3.5-122B-A10B-INT4 + DFlash-8，响应与预填充更强、峰值 tg 略低，是更稳健的次优选择。
    \item \textbf{首字延迟 TTFT}：最低的 TTFT 是 122B-A10B MoE，d0/c1 落在 1.13--1.28s，2200--2421 t/s 的预填充吞吐，约为 27B 稠密档（400--1000 t/s）的 2--5 倍，27B 中仅 NVFP4 的 pp 能上探 ~1900--2000，TTFT 最低也要 1.34s。
    \item \textbf{解码吞吐 TPS}：122B-A10B 的 tg 在 c4 并发普遍 75--92 t/s 且深上文衰减平缓（d4096/c4 80、d8192/c4 52 t/s）；27B 稠密模型中仅 vLLM.build + MTP-4（d0/c4 91.3）、NVFP4 + DFlash-8（87.5）能短暂逼近，但深上文高并发下回落更快。
    \item \textbf{低延迟}：vLLM.dflash + Q3.6-27B-NVFP4 + DFlash-8 —— TTFT 低至 1.34s、pp 峰值 1991 t/s，但 tg 在 30--88 间剧烈波动、响应仅 37，深上文高并发衰减明显，适合高 QPS 短输出。
    \item \textbf{投机解码}：草稿步数并非越大越好，122B 上 MTP-2 / DFlash-2（响应 48）最佳，加大到 MTP-8 / DFlash-15 后 tg 与响应回落（响应降至 40）且质量骤降（MTP-8 质量 73）；27B 上 vLLM.dev MTP-8 质量跌至 53、vLLM.build DFlash-2/4/8 质量仅 43，DFlash-8 是 122B 兼顾 TTFT、TPS 与质量的较优解，过深步数引发质量反噬。
    \item \textbf{DeepSeek-V4-Flash 的 MTP 反噬}：DeepSeek-V4-Flash-IQ2XXS 单跑尚可（pp ~1100、tg 峰值 81.9、质量 90），叠加 MTP-2 后预填充崩塌至 133--365 t/s、tg 跌至 15--21 t/s、质量降至 83、TTFT 同步抬高（d0/c1 5.85s），在 DGX Spark GB10 硬件上为显著负优化。
    \item \textbf{FP8 量化模型}：vLLM.dflash + Q3.6-27B-FP8 + DFlash-15 虽质量 100，却随并发与深度急剧劣化——pp 由 d0/c1 的 1746 t/s 跌到 d8192/c4 的 342 t/s，TTFT 同步爆炸至 119s（d8192/c4），稳定性最差，不宜高并发长上文。
    \item \textbf{超长 128K 上文}：vLLM.dev 122B-A10B MTP-2 的 tg（c1/c2/c4 = 32.4 / 25.2 / 23.1）在多并发下保持力远优于 Hybrid-INT4FP8（33.4 / 13.9 / 10.7），两者 pp ~1100、TTFT 仅 107--275ms，极长上文且多并发时 A10B 更稳。
    \item \textbf{小体积}：LLaMA.cpp + Q3.6-27B（Q4\_K\_M / UD-Q4\_K\_XL）显存占用仅 38--41GiB（约为其余方案的 40\%），MTP-8 档质量满分 100，但 tg 仅 ~18--28 t/s、TTFT 较高（d0/c1 ~2.8s）且深上文 / 并发下进一步走低，适合显存受限、吞吐要求不高的部署。
\end{itemize}
